### 后端接口返回基本格式
```ts
interface BaseResponse<T = any> {
  code: number; /** 状态码 */
  data: T; /** 响应数据 */
  msg: string; /** 信息 */
  page: { /** 分页相关 */
    pageNo: number; /** 当前页 */
    pageSize: number; /** 每页条数 */
    pages: number; /** 总页数 */
    total: number; /** 总条数 */
  };
}
```
